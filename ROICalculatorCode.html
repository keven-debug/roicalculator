function sl(id,outId,fmt){

  const u=()=>out.textContent=fmt(+el.value);

}

function computeROI(){

  const buyers=+$('buyers').value;

  const salary=+$('salary').value;

  const lines=+$('po_lines').value;

  // Block 1: buyer productivity (50% time save)

  const b1_base=buyers*salary*0.5;

  const b1_lo=b1_base*0.75, b1_hi=b1_base*1.0;

  // Block 2: late orders (10% late, $avg line val = lines*0.1, 20% cost, 67% reduction)

  const avgLineVal=850;

  const lateCost=lines*0.10*avgLineVal*0.20;

  const b2_lo=lateCost*0.67*0.5, b2_hi=lateCost*0.67*1.0;

  // Block 3: cross-dept sync (15 non-buyers * 1hr/wk * 52 * hrRate * 85%)

  const nonBuyers=Math.round(buyers*4);

  const hrRate=salary/2080;

  const b3_base=nonBuyers*1*52*hrRate*0.85;

  const b3_lo=b3_base*0.5, b3_hi=b3_base*1.0;

  // Block 4: BI/database cost

function setStep(n){

    $('sc'+i).className='step-circle'+(done?' done':active?' active':'');

    $('sl'+i).className='step-label'+(done?' done':active?' active':'');

    if(i<3)$('sline'+i).className='step-line'+(i<n?' done':'');

  });

  window.scrollTo({top:0,behavior:'smooth'});

}
